69  template<class MetricSet>
70  void read_metrics(std::istream& in, MetricSet& metrics)
71  {
72  typedef typename MetricSet::metric_type metric_type;
73  typedef metric_format_factory <metric_type> factory_type;
74  typedef typename factory_type::metric_format_map metric_format_map;
75  metric_format_map& format_map=factory_type::metric_formats();
76 
77  const int version = in.get();
78  if (format_map.find(version) == format_map.end())
79  throw bad_format_exception("No format found to parse file with version: " +
80  util::lexical_cast<std::string>(version) + " of " +
81  util::lexical_cast<std::string>(format_map.size()));
82  assert(format_map[version]);
83  metrics.set_version(static_cast< ::int16_t>(version));
84  const std::streamsize record_size = format_map[version]->read_header(in, metrics);
85 
86  std::map<typename metric_type::id_t, size_t> metric_offset_map;
87  while(in)
88  {
89  metric_type metric;// Moving this out breaks index metrics - TODO: fix by adding clear to index metrics (or move) or swap
90  const std::streamsize icount = format_map[version]->read_metric_id(in, metric);
91  if(in.fail())
92  {
93  if(icount==0 && !metric_offset_map.empty()) break;
94  throw incomplete_file_exception(std::string("Insufficient data read from the file, got: ")+
95  util::lexical_cast<std::string>(icount)+" != expected: "+
96  util::lexical_cast<std::string>(record_size));
97  }
98 
99  if (metric_offset_map.find(metric.id()) == metric_offset_map.end())
100  {
101  const std::streamsize count = format_map[version]->read_metric(in, metric, metrics, true)+icount;
102  if(in.fail())
103  {
104  if(count==0 && !metric_offset_map.empty()) break;
105  throw incomplete_file_exception(std::string("Insufficient data read from the file, got: ")+
106  util::lexical_cast<std::string>(count)+" != expected: "+
107  util::lexical_cast<std::string>(record_size));
108  }
109  if(count != record_size)
110  {
111  throw bad_format_exception("Record does not match expected size!");
112  }
113  if(metric.lane() > 0 && metric.tile() > 0)
114  {
115  metric_offset_map[metric.id()] = metrics.size();
116  metrics.insert(metric.id(), metric);
117  metrics.metric_updated_at(metric_offset_map[metric.id()]);
118  }
119  }
120  else
121  {
122  const size_t offset = metric_offset_map[metric.id()];
123  INTEROP_ASSERTMSG(metrics.at(offset).lane() != 0, offset);
124  const std::streamsize count = format_map[version]->read_metric(in, metrics.at(offset), metrics, false)+icount;
125  if(in.fail())
126  {
127  if(count==0 && !metric_offset_map.empty()) break;
128  throw incomplete_file_exception(std::string("Insufficient data read from the file, got: ")+
129  util::lexical_cast<std::string>(count)+" != expected: "+
130  util::lexical_cast<std::string>(record_size));
131  }
132  if(count != record_size)
133  {
134  throw bad_format_exception("Record does not match expected size!");
135  }
136  metrics.metric_updated_at(offset);
137  }
138  }
139  }
